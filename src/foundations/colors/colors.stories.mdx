import { Meta, Story, Canvas } from '@storybook/addon-docs/blocks';
import styled from 'styled-components';

import GlobalStyle from '../../global-styles/globalStyles';
import Shape from '../shape';

import { createColorShades } from '../../utils';

<Meta title="Foundations/Palette" />

export const Grid = styled.section`
    display: flex;
    flex-direction: row;
    flex: 1;
    padding: 12px;
`;

export const Col = styled.section`
    display: flex;
    flex-direction: column;
    flex: 1;
`;

export const argTypes = {
    colors: {
        control: null,
    },
};

export const defaultArgs = {
    borderRadius: 'none',
    elevation: 0,
    component: 'div',
    width: 128,
    height: 64,
};

export const Template = args => (
    <>
        <GlobalStyle />
        {args.colors &&
            Object.keys(args.colors).map(shade => {
                return <Shape {...defaultArgs} background={args.colors[shade]} />;
            })}
    </>
);

<Canvas>
    <Story
        name="Blue"
        argTypes={{}}
        args={{
            colors: createColorShades('#1C58D9'),
        }}
    >
        {Template.bind({})}
    </Story>
    <Story
        name="Yellow"
        argTypes={{}}
        args={{
            colors: createColorShades('#7A5600'),
        }}
    >
        {Template.bind({})}
    </Story>
</Canvas>
