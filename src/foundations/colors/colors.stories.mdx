import { Meta, Story, Canvas } from '@storybook/addon-docs/blocks';
import styled from 'styled-components';

import GlobalStyle from '../../global-styles/globalStyles';
import Shape from '../shape';
import Grid from '../layout/Grid';

import { createColorShades } from '../../utils';
import Swatch from './Swatch';

<Meta title="Foundations/Palette" />

export const argTypes = {
    colors: {
        control: null,
    },
};

export const defaultArgs = {
    borderRadius: 'none',
    elevation: 0,
    component: 'div',
    width: 128,
    height: 64,
};

export const Template = args => {
    const colors1 = createColorShades(args.color1);
    const colors2 = args.color2 ? createColorShades(args.color2) : null;
    return (
        <>
            <GlobalStyle />
            <Grid row>
                <Grid>
                    {colors1 &&
                        Object.keys(colors1).map(shade => <Swatch key={`color1_${shade}`} color={colors1[shade]} />)}
                    )}
                </Grid>
                <Grid>
                    {colors2 &&
                        Object.keys(colors2).map(shade => <Swatch key={`color1_${shade}`} color={colors2[shade]} />)}
                    )}
                </Grid>
            </Grid>
        </>
    );
};

<Canvas>
    <Story
        name="Custom Color"
        argTypes={{
            color1: { control: 'color' },
            color2: { control: 'color' },
        }}
        args={{
            color1: '#1C58D9',
            color2: '#7A5600',
        }}
    >
        {Template.bind({})}
    </Story>
    <Story
        name="Blue"
        argTypes={argTypes}
        args={{
            color1: '#1C58D9',
        }}
    >
        {Template.bind({})}
    </Story>
    <Story
        name="Yellow"
        argTypes={argTypes}
        args={{
            color1: '#7A5600',
        }}
    >
        {Template.bind({})}
    </Story>
</Canvas>
